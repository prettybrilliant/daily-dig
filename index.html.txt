function fetchDailyText(){
  return new Promise(function(resolve,reject){
    var now=new Date();

      if(html&&html.length>100){

function parseDailyText(html,date){
  try{
    var doc=new DOMParser().parseFromString(html,'text/html');
    var verse='',body='';
    var themeEl=doc.querySelector('.themeScrp');
    if(themeEl)verse=themeEl.textContent.trim();
    var articleEl=doc.querySelector('#article');
    if(articleEl){
      var cl=articleEl.cloneNode(true);
      var tc=cl.querySelector('.themeScrp');
      if(tc)tc.parentNode.removeChild(tc);
      var ps=cl.querySelectorAll('p'),parts=[];
      for(var i=0;i<ps.length;i++){var t=ps[i].textContent.trim();if(t.length>20)parts.push(t);}
      if(parts.length>0)body=parts.join(' ');
    }
    if(!body){var be=doc.querySelector('.bodyTxt');if(be)body=be.textContent.trim();}
    if(!verse||!body){
      var els=doc.querySelectorAll('h1,h2,h3,p,blockquote'),found=false,cands=[];
      for(var j=0;j<els.length;j++){
        var txt=els[j].textContent.trim();
        if(!found&&txt.match(/january|february|march|april|may|june|july|august|september|october|november|december/i)&&txt.length<60){found=true;continue;}
        if(found&&txt.length>15)cands.push(txt);
      }
      if(cands.length>0&&!verse)verse=cands[0];
      if(cands.length>1&&!body)body=cands.slice(1).join(' ');
    }
    if(!verse||!body){
      var ap=doc.querySelectorAll('p'),lt=[];
      for(var k=0;k<ap.length;k++){var pt=ap[k].textContent.trim();if(pt.length>40)lt.push(pt);}
      if(lt.length>0&&!verse)verse=lt[0];
      if(lt.length>1&&!body)body=lt.slice(1,6).join(' ');
    }
    verse=verse.replace(/\s+/g,' ').trim();
    body=body.replace(/\s+/g,' ').trim();
    if(!verse&&!body)return null;
    return{dateLabel:date.toLocaleDateString('en-US',{weekday:'long',month:'long',day:'numeric'}),verse:verse||'(verse unavailable)',body:body||'(text unavailable)'};
  }catch(e){return null;}
}
